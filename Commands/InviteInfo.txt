    function errorMessage(error) {

        let errorTemplate = {
            color: 0xFF0000,
            title: `Invite Information`,
            description: error,
            timestamp: new Date(),
        }

        msg.channel.send({
            embed: errorTemplate
        });
    }

    function until(conditionFunction) {

        const poll = resolve => {
            if (conditionFunction()) resolve();
            else setTimeout(_ => poll(resolve), 400);
        }

        return new Promise(poll);
    }
